59      /**
60       * Converts the {@link Throwable} to a {@link Status}, taking into account exceptions specific to Armeria as
61       * well and the protocol package.
62       */
63      public static Status fromThrowable(Throwable t) {
64          requireNonNull(t, "t");
65  
66          Throwable cause = t;
67          while (cause != null) {
68              if (cause instanceof ArmeriaStatusException) {
69                  t = StatusExceptionConverter.toGrpc((ArmeriaStatusException) cause);
70                  break;
71              }
72              cause = cause.getCause();
73          }
74  
75          final Status s = Status.fromThrowable(t);
76          if (s.getCode() != Code.UNKNOWN) {
77              return s;
78          }
79          if (t instanceof StreamException) {
80              final StreamException streamException = (StreamException) t;
81              if (streamException.getMessage() != null && streamException.getMessage().contains("RST_STREAM")) {
82                  return Status.CANCELLED;
83              }
84          }
85          if (t instanceof ClosedChannelException) {
86              // ClosedChannelException is used any time the Netty channel is closed. Proper error
87              // processing requires remembering the error that occurred before this one and using it
88              // instead.
89              return Status.UNKNOWN.withCause(t);
90          }
91          if (t instanceof IOException) {
92              return Status.UNAVAILABLE.withCause(t);
93          }
94          if (t instanceof Http2Exception) {
95              return Status.INTERNAL.withCause(t);
96          }
97          if (t instanceof TimeoutException) {
98              return Status.DEADLINE_EXCEEDED.withCause(t);
99          }
100         return s;
101     }

103     /**
104      * Maps GRPC status codes to http status, as defined in upstream grpc-gateway
105      * <a href="https://github.com/grpc-ecosystem/grpc-gateway/blob/master/third_party/googleapis/google/rpc/code.proto">code.proto</a>.
106      */
107     public static HttpStatus grpcCodeToHttpStatus(Status.Code grpcStatusCode) {
108         switch (grpcStatusCode) {
109             case OK:
110                 return HttpStatus.OK;
111             case CANCELLED:
112                 return HttpStatus.CLIENT_CLOSED_REQUEST;
113             case UNKNOWN:
114             case INTERNAL:
115             case DATA_LOSS:
116                 return HttpStatus.INTERNAL_SERVER_ERROR;
117             case INVALID_ARGUMENT:
118             case FAILED_PRECONDITION:
119             case OUT_OF_RANGE:
120                 return HttpStatus.BAD_REQUEST;
121             case DEADLINE_EXCEEDED:
122                 return HttpStatus.GATEWAY_TIMEOUT;
123             case NOT_FOUND:
124                 return HttpStatus.NOT_FOUND;
125             case ALREADY_EXISTS:
126             case ABORTED:
127                 return HttpStatus.CONFLICT;
128             case PERMISSION_DENIED:
129                 return HttpStatus.FORBIDDEN;
130             case UNAUTHENTICATED:
131                 return HttpStatus.UNAUTHORIZED;
132             case RESOURCE_EXHAUSTED:
133                 return HttpStatus.TOO_MANY_REQUESTS;
134             case UNIMPLEMENTED:
135                 return HttpStatus.NOT_IMPLEMENTED;
136             case UNAVAILABLE:
137                 return HttpStatus.SERVICE_UNAVAILABLE;
138             default:
139                 return HttpStatus.UNKNOWN;
140         }
141     }
142 
143     /**
144      * Maps HTTP error response status codes to transport codes, as defined in <a
145      * href="https://github.com/grpc/grpc/blob/master/doc/http-grpc-status-mapping.md">
146      * http-grpc-status-mapping.md</a>. Never returns a status for which {@code status.isOk()} is
147      * {@code true}.
148      *
149      * <p>Copied from <a href="https://github.com/grpc/grpc-java/blob/master/core/src/main/java/io/grpc/internal/GrpcUtil.java">
150      * GrpcUtil.java</a>
151      */
152     public static Status httpStatusToGrpcStatus(int httpStatusCode) {
153         return httpStatusToGrpcCode(httpStatusCode).toStatus()
154                                                    .withDescription("HTTP status code " + httpStatusCode);
155     }
156 
157     private static Status.Code httpStatusToGrpcCode(int httpStatusCode) {
158         if (httpStatusCode >= 100 && httpStatusCode < 200) {
159             // 1xx. These headers should have been ignored.
160             return Status.Code.INTERNAL;
161         }
162         switch (httpStatusCode) {
163             case HttpURLConnection.HTTP_BAD_REQUEST:  // 400
164             case 431: // Request Header Fields Too Large
165                 return Status.Code.INTERNAL;
166             case HttpURLConnection.HTTP_UNAUTHORIZED:  // 401
167                 return Status.Code.UNAUTHENTICATED;
168             case HttpURLConnection.HTTP_FORBIDDEN:  // 403
169                 return Status.Code.PERMISSION_DENIED;
170             case HttpURLConnection.HTTP_NOT_FOUND:  // 404
171                 return Status.Code.UNIMPLEMENTED;
172             case 429:  // Too Many Requests
173             case HttpURLConnection.HTTP_BAD_GATEWAY:  // 502
174             case HttpURLConnection.HTTP_UNAVAILABLE:  // 503
175             case HttpURLConnection.HTTP_GATEWAY_TIMEOUT:  // 504
176                 return Status.Code.UNAVAILABLE;
177             default:
178                 return Status.Code.UNKNOWN;
179         }
180     }

182     /**
183      * Fills the information from the {@link Throwable} into a {@link ThrowableProto} for
184      * returning to a client.
185      */
186     public static ThrowableProto serializeThrowable(Throwable t) {
187         final ThrowableProto.Builder builder = ThrowableProto.newBuilder();
188 
189         if (t instanceof StatusCauseException) {
190             final StatusCauseException statusCause = (StatusCauseException) t;
191             builder.setOriginalClassName(statusCause.getOriginalClassName());
192             builder.setOriginalMessage(statusCause.getOriginalMessage());
193         } else {
194             builder.setOriginalClassName(t.getClass().getCanonicalName());
195             builder.setOriginalMessage(Strings.nullToEmpty(t.getMessage()));
196         }
197 
198         for (StackTraceElement element : t.getStackTrace()) {
199             builder.addStackTrace(serializeStackTraceElement(element));
200         }
201 
202         if (t.getCause() != null) {
203             builder.setCause(serializeThrowable(t.getCause()));
204         }
205         return builder.build();
206     }
207 
208     private static StackTraceElementProto serializeStackTraceElement(StackTraceElement element) {
209         final StackTraceElementProto.Builder builder =
210                 StackTraceElementProto.newBuilder()
211                                       .setClassName(element.getClassName())
212                                       .setMethodName(element.getMethodName())
213                                       .setLineNumber(element.getLineNumber());
214         if (element.getFileName() != null) {
215             builder.setFileName(element.getFileName());
216         }
217         return builder.build();
218     }
