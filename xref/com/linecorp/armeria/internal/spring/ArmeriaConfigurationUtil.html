119     /**
120      * Sets graceful shutdown timeout, health check services and {@link MeterRegistry} for the specified
121      * {@link ServerBuilder}.
122      */
123     public static void configureServerWithArmeriaSettings(ServerBuilder server, ArmeriaSettings settings,
124                                                           MeterRegistry meterRegistry,
125                                                           List<HealthChecker> healthCheckers) {
126         requireNonNull(server, "server");
127         requireNonNull(settings, "settings");
128         requireNonNull(meterRegistry, "meterRegistry");
129         requireNonNull(healthCheckers, "healthCheckers");
130 
131         if (settings.getGracefulShutdownQuietPeriodMillis() >= 0 &&
132             settings.getGracefulShutdownTimeoutMillis() >= 0) {
133             server.gracefulShutdownTimeout(settings.getGracefulShutdownQuietPeriodMillis(),
134                                            settings.getGracefulShutdownTimeoutMillis());
135             logger.debug("Set graceful shutdown timeout: quiet period {} ms, timeout {} ms",
136                          settings.getGracefulShutdownQuietPeriodMillis(),
137                          settings.getGracefulShutdownTimeoutMillis());
138         }
139 
140         final String healthCheckPath = settings.getHealthCheckPath();
141         if (!Strings.isNullOrEmpty(healthCheckPath)) {
142             server.service(healthCheckPath,
143                            new HttpHealthCheckService(healthCheckers.toArray(EMPTY_HEALTH_CHECKERS)));
144         }
145 
146         server.meterRegistry(meterRegistry);
147 
148         if (settings.isEnableMetrics() && !Strings.isNullOrEmpty(settings.getMetricsPath())) {
149             if (meterRegistry instanceof CompositeMeterRegistry) {
150                 final Set<MeterRegistry> childRegistries =
151                         ((CompositeMeterRegistry) meterRegistry).getRegistries();
152                 childRegistries.stream()
153                                .filter(PrometheusMeterRegistry.class::isInstance)
154                                .map(PrometheusMeterRegistry.class::cast)
155                                .findAny()
156                                .ifPresent(r -> addPrometheusExposition(settings, server, r));
157             } else if (meterRegistry instanceof PrometheusMeterRegistry) {
158                 addPrometheusExposition(settings, server, (PrometheusMeterRegistry) meterRegistry);
159             } else if (meterRegistry instanceof DropwizardMeterRegistry) {
160                 final MetricRegistry dropwizardRegistry =
161                         ((DropwizardMeterRegistry) meterRegistry).getDropwizardRegistry();
162                 final ObjectMapper objectMapper = new ObjectMapper()
163                         .enable(SerializationFeature.INDENT_OUTPUT)
164                         .registerModule(new MetricsModule(TimeUnit.SECONDS, TimeUnit.MILLISECONDS, true));
165                 server.service(
166                         settings.getMetricsPath(),
167                         new AbstractHttpService() {
168                             @Override
169                             protected HttpResponse doGet(ServiceRequestContext ctx, HttpRequest req)
170                                     throws Exception {
171                                 return HttpResponse.of(HttpStatus.OK, MediaType.JSON_UTF_8,
172                                                        objectMapper.writeValueAsBytes(dropwizardRegistry));
173                             }
174                         });
175             }
176         }
177 
178         if (settings.getSsl() != null) {
179             configureTls(server, settings.getSsl());
180         }
181 
182         final ArmeriaSettings.Compression compression = settings.getCompression();
183         if (compression != null && compression.isEnabled()) {
184             server.decorator(contentEncodingDecorator(compression.getMimeTypes(),
185                                                       compression.getExcludedUserAgents(),
186                                                       parseDataSize(compression.getMinResponseSize())));
187         }
188     }

190     /**
191      * Adds {@link Port}s to the specified {@link ServerBuilder}.
192      */
193     public static void configurePorts(ServerBuilder server, List<Port> ports) {
194         requireNonNull(server, "server");
195         requireNonNull(ports, "ports");
196         ports.forEach(p -> {
197             final String ip = p.getIp();
198             final String iface = p.getIface();
199             final int port = p.getPort();
200             final List<SessionProtocol> protocols = p.getProtocols();
201 
202             if (ip == null) {
203                 if (iface == null) {
204                     server.port(new ServerPort(port, protocols));
205                 } else {
206                     try {
207                         final Enumeration<InetAddress> e = NetworkInterface.getByName(iface).getInetAddresses();
208                         while (e.hasMoreElements()) {
209                             server.port(new ServerPort(new InetSocketAddress(e.nextElement(), port),
210                                                        protocols));
211                         }
212                     } catch (SocketException e) {
213                         throw new IllegalStateException("Failed to find an iface: " + iface, e);
214                     }
215                 }
216             } else if (iface == null) {
217                 if (NetUtil.isValidIpV4Address(ip) || NetUtil.isValidIpV6Address(ip)) {
218                     final byte[] bytes = NetUtil.createByteArrayFromIpAddressString(ip);
219                     try {
220                         server.port(new ServerPort(new InetSocketAddress(
221                                 InetAddress.getByAddress(bytes), port), protocols));
222                     } catch (UnknownHostException e) {
223                         // Should never happen.
224                         throw new Error(e);
225                     }
226                 } else {
227                     throw new IllegalStateException("invalid IP address: " + ip);
228                 }
229             } else {
230                 throw new IllegalStateException("A port cannot have both IP and iface: " + p);
231             }
232         });
233     }

235     /**
236      * Adds Thrift services to the specified {@link ServerBuilder}.
237      */
238     public static void configureThriftServices(
239             ServerBuilder server, List<ThriftServiceRegistrationBean> beans,
240             @Nullable MeterIdPrefixFunctionFactory meterIdPrefixFunctionFactory,
241             @Nullable String docsPath) {
242         requireNonNull(server, "server");
243         requireNonNull(beans, "beans");
244 
245         final List<TBase<?, ?>> docServiceRequests = new ArrayList<>();
246         final Map<String, Collection<HttpHeaders>> docServiceHeaders = new HashMap<>();
247         beans.forEach(bean -> {
248             Service<HttpRequest, HttpResponse> service = bean.getService();
249             for (Function<Service<HttpRequest, HttpResponse>, ? extends Service<HttpRequest, HttpResponse>>
250                     decorator : bean.getDecorators()) {
251                 service = service.decorate(decorator);
252             }
253             service = setupMetricCollectingService(service, bean, meterIdPrefixFunctionFactory);
254             server.service(bean.getPath(), service);
255             docServiceRequests.addAll(bean.getExampleRequests());
256             ThriftServiceUtils.serviceNames(bean.getService())
257                               .forEach(serviceName -> docServiceHeaders
258                                       .put(serviceName,
259                                            bean.getExampleHeaders()));
260         });
261 
262         if (!Strings.isNullOrEmpty(docsPath)) {
263             final DocServiceBuilder docServiceBuilder = new DocServiceBuilder();
264             docServiceBuilder.exampleRequest(docServiceRequests);
265             for (Entry<String, Collection<HttpHeaders>> entry : docServiceHeaders.entrySet()) {
266                 docServiceBuilder.exampleHttpHeaders(entry.getKey(), entry.getValue());
267             }
268             server.serviceUnder(docsPath, docServiceBuilder.build());
269         }
270     }

275     public static void configureHttpServices(
276             ServerBuilder server, List<HttpServiceRegistrationBean> beans,
277             @Nullable MeterIdPrefixFunctionFactory meterIdPrefixFunctionFactory) {
278         requireNonNull(server, "server");
279         requireNonNull(beans, "beans");
280 
281         beans.forEach(bean -> {
282             Service<HttpRequest, HttpResponse> service = bean.getService();
283             for (Function<Service<HttpRequest, HttpResponse>, ? extends Service<HttpRequest, HttpResponse>>
284                     decorator : bean.getDecorators()) {
285                 service = service.decorate(decorator);
286             }
287             service = setupMetricCollectingService(service, bean, meterIdPrefixFunctionFactory);
288             server.service(bean.getPathMapping(), service);
289         });
290     }

292     /**
293      * Adds annotated HTTP services to the specified {@link ServerBuilder}.
294      */
295     public static void configureAnnotatedHttpServices(
296             ServerBuilder server, List<AnnotatedServiceRegistrationBean> beans,
297             @Nullable MeterIdPrefixFunctionFactory meterIdPrefixFunctionFactory) {
298         requireNonNull(server, "server");
299         requireNonNull(beans, "beans");
300 
301         beans.forEach(bean -> {
302             Function<Service<HttpRequest, HttpResponse>,
303                     ? extends Service<HttpRequest, HttpResponse>> decorator = Function.identity();
304             for (Function<Service<HttpRequest, HttpResponse>, ? extends Service<HttpRequest, HttpResponse>>
305                     d : bean.getDecorators()) {
306                 decorator = decorator.andThen(d);
307             }
308             if (meterIdPrefixFunctionFactory != null) {
309                 decorator = decorator.andThen(
310                         metricCollectingServiceDecorator(bean, meterIdPrefixFunctionFactory));
311             }
312             final ImmutableList<Object> exceptionHandlersAndConverters =
313                     ImmutableList.builder()
314                                  .addAll(bean.getExceptionHandlers())
315                                  .addAll(bean.getRequestConverters())
316                                  .addAll(bean.getResponseConverters())
317                                  .build();
318             server.annotatedService(bean.getPathPrefix(), bean.getService(), decorator,
319                                     exceptionHandlersAndConverters);
320         });
321     }

323     private static Service<HttpRequest, HttpResponse> setupMetricCollectingService(
324             Service<HttpRequest, HttpResponse> service,
325             AbstractServiceRegistrationBean<?, ?> bean,
326             @Nullable MeterIdPrefixFunctionFactory meterIdPrefixFunctionFactory) {
327         requireNonNull(service, "service");
328         requireNonNull(bean, "bean");
329 
330         if (meterIdPrefixFunctionFactory == null) {
331             return service;
332         }
333         return service.decorate(metricCollectingServiceDecorator(bean, meterIdPrefixFunctionFactory));
334     }

337             MetricCollectingService<HttpRequest, HttpResponse>> metricCollectingServiceDecorator(
338             AbstractServiceRegistrationBean<?, ?> bean,
339             MeterIdPrefixFunctionFactory meterIdPrefixFunctionFactory) {
340         requireNonNull(bean, "bean");
341         requireNonNull(meterIdPrefixFunctionFactory, "meterIdPrefixFunctionFactory");

347     private static void addPrometheusExposition(ArmeriaSettings armeriaSettings, ServerBuilder server,
348                                                 PrometheusMeterRegistry registry) {
349         requireNonNull(armeriaSettings, "armeriaSettings");
350         requireNonNull(server, "server");
351         requireNonNull(registry, "registry");

369     /**
370      * Adds SSL/TLS context to the specified {@link ServerBuilder}.
371      */
372     public static void configureTls(ServerBuilder sb, Ssl ssl,
373                                     @Nullable Supplier<KeyStore> keyStoreSupplier,
374                                     @Nullable Supplier<KeyStore> trustStoreSupplier) {
375         if (!ssl.isEnabled()) {
376             return;
377         }
378         try {
379             if (keyStoreSupplier == null && trustStoreSupplier == null &&
380                 ssl.getKeyStore() == null && ssl.getTrustStore() == null) {
381                 logger.warn("Configuring TLS with a self-signed certificate " +
382                             "because no key or trust store was specified");
383                 sb.tlsSelfSigned();
384                 return;
385             }
386 
387             final KeyManagerFactory keyManagerFactory = getKeyManagerFactory(ssl, keyStoreSupplier);
388             final TrustManagerFactory trustManagerFactory = getTrustManagerFactory(ssl, trustStoreSupplier);
389 
390             sb.tls(keyManagerFactory, sslContextBuilder -> {
391                 sslContextBuilder.trustManager(trustManagerFactory);
392 
393                 final SslProvider sslProvider = ssl.getProvider();
394                 if (sslProvider != null) {
395                     sslContextBuilder.sslProvider(sslProvider);
396                 }
397                 final List<String> enabledProtocols = ssl.getEnabledProtocols();
398                 if (enabledProtocols != null) {
399                     sslContextBuilder.protocols(enabledProtocols.toArray(new String[enabledProtocols.size()]));
400                 }
401                 final List<String> ciphers = ssl.getCiphers();
402                 if (ciphers != null) {
403                     sslContextBuilder.ciphers(ImmutableList.copyOf(ciphers),
404                                               SupportedCipherSuiteFilter.INSTANCE);
405                 }
406                 final ClientAuth clientAuth = ssl.getClientAuth();
407                 if (clientAuth != null) {
408                     sslContextBuilder.clientAuth(clientAuth);
409                 }
410             });
411         } catch (Exception e) {
412             throw new IllegalStateException("Failed to configure TLS: " + e, e);
413         }
414     }

416     private static KeyManagerFactory getKeyManagerFactory(
417             Ssl ssl, @Nullable Supplier<KeyStore> sslStoreProvider) throws Exception {
418         final KeyStore store;
419         if (sslStoreProvider != null) {
420             store = sslStoreProvider.get();
421         } else {
422             store = loadKeyStore(ssl.getKeyStoreType(), ssl.getKeyStore(), ssl.getKeyStorePassword());
423         }
424 
425         final KeyManagerFactory keyManagerFactory =
426                 KeyManagerFactory.getInstance(KeyManagerFactory.getDefaultAlgorithm());
427 
428         String keyPassword = ssl.getKeyPassword();
429         if (keyPassword == null) {
430             keyPassword = ssl.getKeyStorePassword();
431         }
432 
433         keyManagerFactory.init(store, keyPassword != null ? keyPassword.toCharArray()
434                                                           : null);
435         return keyManagerFactory;
436     }
437 
438     private static TrustManagerFactory getTrustManagerFactory(
439             Ssl ssl, @Nullable Supplier<KeyStore> sslStoreProvider) throws Exception {
440         final KeyStore store;
441         if (sslStoreProvider != null) {
442             store = sslStoreProvider.get();
443         } else {
444             store = loadKeyStore(ssl.getTrustStoreType(), ssl.getTrustStore(), ssl.getTrustStorePassword());
445         }
446 
447         final TrustManagerFactory trustManagerFactory =
448                 TrustManagerFactory.getInstance(TrustManagerFactory.getDefaultAlgorithm());
449         trustManagerFactory.init(store);
450         return trustManagerFactory;
451     }
452 
453     @Nullable
454     private static KeyStore loadKeyStore(
455             @Nullable String type,
456             @Nullable String resource,
457             @Nullable String password) throws IOException, GeneralSecurityException {
458         if (resource == null) {
459             return null;
460         }
461         final KeyStore store = KeyStore.getInstance(firstNonNull(type, "JKS"));
462         final URL url = ResourceUtils.getURL(resource);
463         store.load(url.openStream(), password != null ? password.toCharArray()
464                                                       : null);
465         return store;
466     }

468     /**
469      * Configures a decorator for encoding the content of the HTTP responses sent from the server.
470      */
471     public static Function<Service<HttpRequest, HttpResponse>,
472             HttpEncodingService> contentEncodingDecorator(@Nullable String[] mimeTypes,
473                                                           @Nullable String[] excludedUserAgents,
474                                                           long minBytesToForceChunkedAndEncoding) {
475         final Predicate<MediaType> encodableContentTypePredicate;
476         if (mimeTypes == null || mimeTypes.length == 0) {
477             encodableContentTypePredicate = contentType -> true;
478         } else {
479             final List<MediaType> encodableContentTypes =
480                     Arrays.stream(mimeTypes).map(MediaType::parse).collect(toImmutableList());
481             encodableContentTypePredicate = contentType ->
482                     encodableContentTypes.stream().anyMatch(contentType::is);
483         }
484 
485         final Predicate<HttpHeaders> encodableRequestHeadersPredicate;
486         if (excludedUserAgents == null || excludedUserAgents.length == 0) {
487             encodableRequestHeadersPredicate = headers -> true;
488         } else {
489             final List<Pattern> patterns =
490                     Arrays.stream(excludedUserAgents).map(Pattern::compile).collect(toImmutableList());
491             encodableRequestHeadersPredicate = headers -> {
492                 // No User-Agent header will be converted to an empty string.
493                 final String userAgent = headers.get(HttpHeaderNames.USER_AGENT, "");
494                 return patterns.stream().noneMatch(pattern -> pattern.matcher(userAgent).matches());
495             };
496         }
497 
498         return delegate -> new HttpEncodingService(delegate,
499                                                    encodableContentTypePredicate,
500                                                    encodableRequestHeadersPredicate,
501                                                    minBytesToForceChunkedAndEncoding);
502     }

504     /**
505      * Parses the data size text as a decimal {@code long}.
506      *
507      * @param dataSizeText the data size text, i.e. {@code 1}, {@code 1B}, {@code 1KB}, {@code 1MB},
508      *                     {@code 1GB} or {@code 1TB}
509      */
510     public static long parseDataSize(String dataSizeText) {
511         requireNonNull(dataSizeText, "text");
512         final Matcher matcher = DATA_SIZE_PATTERN.matcher(dataSizeText);
513         checkArgument(matcher.matches(),
514                       "Invalid data size text: %s (expected: %s)",
515                       dataSizeText, DATA_SIZE_PATTERN);
516 
517         final long unit;
518         final String unitText = matcher.group(2);
519         if (Strings.isNullOrEmpty(unitText)) {
520             unit = 1L;
521         } else {
522             switch (Ascii.toLowerCase(unitText)) {
523                 case "b":
524                     unit = 1L;
525                     break;
526                 case "kb":
527                     unit = 1024L;
528                     break;
529                 case "mb":
530                     unit = 1024L * 1024L;
531                     break;
532                 case "gb":
533                     unit = 1024L * 1024L * 1024L;
534                     break;
535                 case "tb":
536                     unit = 1024L * 1024L * 1024L * 1024L;
537                     break;
538                 default:
539                     throw new IllegalArgumentException("Invalid data size text: " + dataSizeText +
540                                                        " (expected: " + DATA_SIZE_PATTERN + ')');
541             }
542         }
543         try {
544             final long amount = Long.parseLong(matcher.group(1));
545             return LongMath.checkedMultiply(amount, unit);
546         } catch (Exception e) {
547             throw new IllegalArgumentException("Invalid data size text: " + dataSizeText +
548                                                " (expected: " + DATA_SIZE_PATTERN + ')', e);
549         }
550     }
