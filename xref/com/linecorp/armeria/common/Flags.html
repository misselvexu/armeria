244     static {
245         if (!isEpollAvailable()) {
246             final Throwable cause = Epoll.unavailabilityCause();
247             if (cause != null) {
248                 logger.info("/dev/epoll not available: {}", Exceptions.peel(cause).toString());
249             } else {
250                 if (HAS_WSLENV) {
251                     logger.info("/dev/epoll not available: WSL not supported");
252                 } else {
253                     logger.info("/dev/epoll not available: ?");
254                 }
255             }
256         } else if (USE_EPOLL) {
257             logger.info("Using /dev/epoll");
258         }
259 
260         if (!OpenSsl.isAvailable()) {
261             final Throwable cause = Exceptions.peel(OpenSsl.unavailabilityCause());
262             logger.info("OpenSSL not available: {}", cause.toString());
263         } else if (USE_OPENSSL) {
264             logger.info("Using OpenSSL: {}, 0x{}",
265                         OpenSsl.versionString(),
266                         Long.toHexString(OpenSsl.version() & 0xFFFFFFFFL));
267         }
268     }
269 
270     private static boolean isEpollAvailable() {
271         // Netty epoll transport does not work with WSL (Windows Sybsystem for Linux) yet.
272         // TODO(trustin): Re-enable on WSL if https://github.com/Microsoft/WSL/issues/1982 is resolved.
273         return Epoll.isAvailable() && !HAS_WSLENV;
274     }

288     /**
289      * Returns whether to log the socket exceptions which are mostly harmless. If enabled, the following
290      * exceptions will be logged:
291      * <ul>
292      *   <li>{@link ClosedChannelException}</li>
293      *   <li>{@link ClosedSessionException}</li>
294      *   <li>{@link IOException} - 'Connection reset/closed/aborted by peer'</li>

297      *   <li>{@link SSLException} - 'SSLEngine closed already'</li>
298      * </ul>
299      *

303      *
304      * <p>This flag is disabled by default.
305      * Specify the {@code -Dcom.linecorp.armeria.verboseSocketExceptions=true} JVM option to enable it.</p>

702     private static Optional<String> caffeineSpec(String name, String defaultValue) {
703         final String spec = get(name, defaultValue, value -> {
704             try {
705                 if (!"off".equals(value)) {
706                     CaffeineSpec.parse(value);
707                 }
708                 return true;
709             } catch (Exception e) {
710                 return false;
711             }
712         });
713         return "off".equals(spec) ? Optional.empty()
714                                   : Optional.of(spec);
715     }

717     private static ExceptionVerbosity exceptionLoggingMode(String name, String defaultValue) {
718         final String mode = getNormalized(name, defaultValue,
719                                           value -> Arrays.stream(ExceptionVerbosity.values())
720                                                          .anyMatch(v -> v.name().equalsIgnoreCase(value)));
721         return ExceptionVerbosity.valueOf(mode.toUpperCase());
722     }

728     private static boolean getBoolean(String name, boolean defaultValue, Predicate<Boolean> validator) {
729         return "true".equals(getNormalized(name, String.valueOf(defaultValue), value -> {
730             if ("true".equals(value)) {
731                 return validator.test(true);
732             }
733 
734             if ("false".equals(value)) {
735                 return validator.test(false);
736             }
737 
738             return false;
739         }));
740     }

764     private static String get(String name, String defaultValue, Predicate<String> validator) {
765         final String fullName = PREFIX + name;
766         final String value = System.getProperty(fullName);
767         if (value == null) {
768             logger.info("{}: {} (default)", fullName, defaultValue);
769             return defaultValue;
770         }
771 
772         if (validator.test(value)) {
773             logger.info("{}: {}", fullName, value);
774             return value;
775         }
776 
777         logger.info("{}: {} (default instead of: {})", fullName, defaultValue, value);
778         return defaultValue;
779     }
780 
781     private static String getNormalized(String name, String defaultValue, Predicate<String> validator) {
782         final String fullName = PREFIX + name;
783         final String value = getLowerCased(fullName);
784         if (value == null) {
785             logger.info("{}: {} (default)", fullName, defaultValue);
786             return defaultValue;
787         }
788 
789         if (validator.test(value)) {
790             logger.info("{}: {}", fullName, value);
791             return value;
792         }
793 
794         logger.info("{}: {} (default instead of: {})", fullName, defaultValue, value);
795         return defaultValue;
796     }
797 
798     @Nullable
799     private static String getLowerCased(String fullName) {
800         String value = System.getProperty(fullName);
801         if (value != null) {
802             value = Ascii.toLowerCase(value);
803         }
804         return value;
805     }
