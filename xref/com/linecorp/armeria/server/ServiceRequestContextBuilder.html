16  package com.linecorp.armeria.server;
17  
18  import static java.util.Objects.requireNonNull;
19  
20  import java.net.InetAddress;
21  import java.util.ArrayList;
22  import java.util.List;
23  import java.util.Optional;
24  import java.util.function.Consumer;
25  
26  import javax.annotation.Nullable;
27  
28  import com.linecorp.armeria.common.AbstractRequestContextBuilder;
29  import com.linecorp.armeria.common.HttpRequest;
30  import com.linecorp.armeria.common.HttpResponse;
31  import com.linecorp.armeria.common.HttpStatus;
32  
33  /**
34   * Builds a new {@link ServiceRequestContext}. Note that it is not usually required to create a new context by
35   * yourself, because Armeria will always provide a context object for you. However, it may be useful in some
36   * cases such as unit testing.
37   */
38  public final class ServiceRequestContextBuilder
39          extends AbstractRequestContextBuilder<ServiceRequestContextBuilder> {
40  
41      /**
42       * A placeholder service to make {@link ServerBuilder} happy.
43       */
44      private static final HttpService fakeService = (ctx, req) -> HttpResponse.of(HttpStatus.METHOD_NOT_ALLOWED);
45  
46      /**
47       * A {@link ServerListener} which rejects an attempt to start a {@link Server}.
48       */
49      private static final ServerListener rejectingListener = new ServerListenerAdapter() {
50          @Override
51          public void serverStarting(Server server) throws Exception {
52              throw new UnsupportedOperationException();
53          }
54      };
55  
56      /**
57       * Returns a new {@link ServiceRequestContextBuilder} created from the specified {@link HttpRequest}.
58       */
59      public static ServiceRequestContextBuilder of(HttpRequest request) {
60          return new ServiceRequestContextBuilder(request);
61      }
62  
63      private final List<Consumer<? super ServerBuilder>> serverConfigurators = new ArrayList<>(4);
64  
65      private Service<HttpRequest, HttpResponse> service = fakeService;
66      @Nullable
67      private PathMappingResult pathMappingResult;
68      @Nullable
69      private ProxiedAddresses proxiedAddresses;
70      @Nullable
71      private InetAddress clientAddress;
72  
73      private ServiceRequestContextBuilder(HttpRequest request) {
74          super(true, request);
75      }
76  
77      /**
78       * Sets the {@link Service} that handles the request. If not set, a dummy {@link Service}, which always
79       * returns a {@code "405 Method Not Allowed"} response, is used.
80       */
81      public ServiceRequestContextBuilder service(Service<HttpRequest, HttpResponse> service) {
82          this.service = requireNonNull(service, "service");
83          return this;
84      }
85  
86      /**
87       * Sets the {@link PathMappingResult} of the request. If not set, it is auto-generated from the request.
88       */
89      public ServiceRequestContextBuilder pathMappingResult(PathMappingResult pathMappingResult) {
90          this.pathMappingResult = requireNonNull(pathMappingResult, "pathMappingResult");
91          return this;
92      }
93  
94      /**
95       * Sets the {@link ProxiedAddresses} of the request.
96       * If not set, {@link ServiceRequestContext#proxiedAddresses()} will return {@code null}.
97       */
98      public ServiceRequestContextBuilder proxiedAddresses(ProxiedAddresses proxiedAddresses) {
99          this.proxiedAddresses = requireNonNull(proxiedAddresses, "proxiedAddresses");
100         return this;
101     }
102 
103     /**
104      * Sets the client address of the request. If not set, {@link ServiceRequestContext#clientAddress()} will
105      * return the same value as {@link ServiceRequestContext#remoteAddress()}.
106      */
107     public ServiceRequestContextBuilder clientAddress(InetAddress clientAddress) {
108         this.clientAddress = requireNonNull(clientAddress, "clientAddress");
109         return this;
110     }
111 
112     /**
113      * Adds the {@link Consumer} that configures the given {@link ServerBuilder}. The {@link Consumer}s added
114      * by thid method will be invoked when this builder builds a dummy {@link Server}. This may be useful
115      * when you need to update the default settings of the dummy {@link Server},
116      * such as {@link ServerConfig#defaultMaxRequestLength()}.
117      */
118     public ServiceRequestContextBuilder serverConfigurator(Consumer<? super ServerBuilder> serverConfigurator) {
119         requireNonNull(serverConfigurator, "serverConfigurator");
120         serverConfigurators.add(serverConfigurator);
121         return this;
122     }
123 
124     /**
125      * Returns a new {@link ServiceRequestContext} created with the properties of this builder.
126      */
127     public ServiceRequestContext build() {
128         // Determine the client address; use remote address unless overridden.
129         final InetAddress clientAddress;
130         if (this.clientAddress != null) {
131             clientAddress = this.clientAddress;
132         } else {
133             clientAddress = remoteAddress().getAddress();
134         }
135 
136         // Build a fake server which never starts up.
137         final ServerBuilder serverBuilder = new ServerBuilder().meterRegistry(meterRegistry())
138                                                                .workerGroup(eventLoop(), false)
139                                                                .service(path(), service);
140         serverConfigurators.forEach(configurator -> configurator.accept(serverBuilder));
141 
142         final Server server = serverBuilder.build();
143         server.addListener(rejectingListener);
144 
145         // Retrieve the ServiceConfig of the fake service.
146         final ServiceConfig serviceCfg = findServiceConfig(server, path(), service);
147 
148         // Build a fake object related with path mapping.
149         final PathMappingContext pathMappingCtx = DefaultPathMappingContext.of(
150                 server.config().defaultVirtualHost(),
151                 localAddress().getHostString(),
152                 path(),
153                 query(),
154                 ((HttpRequest) request()).headers(),
155                 null);
156 
157         final PathMappingResult pathMappingResult =
158                 this.pathMappingResult != null ? this.pathMappingResult
159                                                : PathMappingResult.of(path(), query());
160 
161         // Build the context with the properties set by a user and the fake objects.
162         if (isRequestStartTimeSet()) {
163             return new DefaultServiceRequestContext(
164                     serviceCfg, fakeChannel(), meterRegistry(), sessionProtocol(), pathMappingCtx,
165                     pathMappingResult, request(), sslSession(), proxiedAddresses, clientAddress,
166                     requestStartTimeNanos(), requestStartTimeMicros());
167         } else {
168             return new DefaultServiceRequestContext(
169                     serviceCfg, fakeChannel(), meterRegistry(), sessionProtocol(), pathMappingCtx,
170                     pathMappingResult, request(), sslSession(), proxiedAddresses, clientAddress);
171         }
172     }
173 
174     private static ServiceConfig findServiceConfig(Server server, String path, Service<?, ?> service) {
175         for (ServiceConfig cfg : server.config().defaultVirtualHost().serviceConfigs()) {
176             final Optional<String> exactPath = cfg.pathMapping().exactPath();
177             if (!exactPath.isPresent()) {
178                 continue;
179             }
180 
181             if (!path.equals(exactPath.get())) {
182                 continue;
183             }
184 
185             if (cfg.service().as(service.getClass()).isPresent()) {
186                 return cfg;
187             }
188         }
189 
190         throw new Error(); // Never reaches here.
191     }
192 }
