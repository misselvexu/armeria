154     private DefaultServiceRequestContext(
155             ServiceConfig cfg, Channel ch, MeterRegistry meterRegistry, SessionProtocol sessionProtocol,
156             PathMappingContext pathMappingContext, PathMappingResult pathMappingResult, HttpRequest request,
157             @Nullable SSLSession sslSession, @Nullable ProxiedAddresses proxiedAddresses,
158             InetAddress clientAddress, boolean requestStartTimeSet, long requestStartTimeNanos,
159             long requestStartTimeMicros) {
160 
161         super(meterRegistry, sessionProtocol,
162               requireNonNull(pathMappingContext, "pathMappingContext").method(), pathMappingContext.path(),
163               requireNonNull(pathMappingResult, "pathMappingResult").query(),
164               request);
165 
166         this.ch = requireNonNull(ch, "ch");
167         this.cfg = requireNonNull(cfg, "cfg");
168         this.pathMappingContext = pathMappingContext;
169         this.pathMappingResult = pathMappingResult;
170         this.sslSession = sslSession;
171         this.proxiedAddresses = proxiedAddresses;
172         this.clientAddress = requireNonNull(clientAddress, "clientAddress");
173 
174         log = new DefaultRequestLog(this, virtualHost().requestContentPreviewerFactory(),
175                                     virtualHost().responseContentPreviewerFactory());
176         if (requestStartTimeSet) {
177             log.startRequest(ch, sessionProtocol, sslSession, requestStartTimeNanos, requestStartTimeMicros);
178         } else {
179             log.startRequest(ch, sessionProtocol, sslSession);
180         }
181         log.requestHeaders(request.headers());
182 
183         // For the server, request headers are processed well before ServiceRequestContext is created. It means
184         // there is some delay between the actual channel read and this logging, but it's the best we can do for
185         // now.
186         log.requestFirstBytesTransferred();
187 
188         logger = newLogger(cfg);
189 
190         final ServerConfig serverCfg = cfg.server().config();
191         requestTimeoutMillis = serverCfg.defaultRequestTimeoutMillis();
192         maxRequestLength = serverCfg.defaultMaxRequestLength();
193     }

195     private RequestContextAwareLogger newLogger(ServiceConfig cfg) {
196         String loggerName = cfg.loggerName().orElse(null);
197         if (loggerName == null) {
198             loggerName = cfg.pathMapping().loggerName();
199         }
200 
201         return new RequestContextAwareLogger(this, LoggerFactory.getLogger(
202                 cfg.server().config().serviceLoggerPrefix() + '.' + loggerName));
203     }

235     @Override
236     public ServiceRequestContext newDerivedContext(Request request) {
237         final DefaultServiceRequestContext ctx = new DefaultServiceRequestContext(
238                 cfg, ch, meterRegistry(), sessionProtocol(), pathMappingContext,
239                 pathMappingResult, (HttpRequest) request, sslSession(), proxiedAddresses(), clientAddress);
240 
241         final HttpHeaders additionalHeaders = additionalResponseHeaders();
242         if (!additionalHeaders.isEmpty()) {
243             ctx.setAdditionalResponseHeaders(additionalHeaders);
244         }
245 
246         final HttpHeaders additionalTrailers = additionalResponseTrailers();
247         if (!additionalTrailers.isEmpty()) {
248             ctx.setAdditionalResponseTrailers(additionalTrailers);
249         }
250 
251         for (final Iterator<Attribute<?>> i = attrs(); i.hasNext();/* noop */) {
252             ctx.addAttr(i.next());
253         }
254         return ctx;
255     }

257     private HttpHeaders createAdditionalHeadersIfAbsent() {
258         final HttpHeaders additionalResponseHeaders = this.additionalResponseHeaders;
259         if (additionalResponseHeaders == null) {
260             final HttpHeaders newHeaders = new DefaultHttpHeaders();
261             if (additionalResponseHeadersUpdater.compareAndSet(this, null, newHeaders)) {
262                 return newHeaders;
263             } else {
264                 final HttpHeaders additionalResponseHeadersSetByOtherThread = this.additionalResponseHeaders;
265                 assert additionalResponseHeadersSetByOtherThread != null;
266                 return additionalResponseHeadersSetByOtherThread;
267             }
268         } else {
269             return additionalResponseHeaders;
270         }
271     }
272 
273     private HttpHeaders createAdditionalTrailersIfAbsent() {
274         final HttpHeaders additionalResponseTrailers = this.additionalResponseTrailers;
275         if (additionalResponseTrailers == null) {
276             final HttpHeaders newTrailers = new DefaultHttpHeaders();
277             if (additionalResponseTrailersUpdater.compareAndSet(this, null, newTrailers)) {
278                 return newTrailers;
279             } else {
280                 final HttpHeaders additionalResponseTrailersSetByOtherThread = this.additionalResponseTrailers;
281                 assert additionalResponseTrailersSetByOtherThread != null;
282                 return additionalResponseTrailersSetByOtherThread;
283             }
284         } else {
285             return additionalResponseTrailers;
286         }
287     }

376     @Override
377     public void setRequestTimeoutMillis(long requestTimeoutMillis) {
378         if (requestTimeoutMillis < 0) {
379             throw new IllegalArgumentException(
380                     "requestTimeoutMillis: " + requestTimeoutMillis + " (expected: >= 0)");
381         }
382         if (this.requestTimeoutMillis != requestTimeoutMillis) {
383             this.requestTimeoutMillis = requestTimeoutMillis;
384             final RequestTimeoutChangeListener listener = requestTimeoutChangeListener;
385             if (listener != null) {
386                 if (ch.eventLoop().inEventLoop()) {
387                     listener.onRequestTimeoutChange(requestTimeoutMillis);
388                 } else {
389                     ch.eventLoop().execute(() -> listener.onRequestTimeoutChange(requestTimeoutMillis));
390                 }
391             }
392         }
393     }

536     /**
537      * Sets the listener that is notified when the {@linkplain #requestTimeoutMillis()} request timeout} of

542      */
543     public void setRequestTimeoutChangeListener(RequestTimeoutChangeListener listener) {
544         requireNonNull(listener, "listener");

548         requestTimeoutChangeListener = listener;

551     @Override
552     public String toString() {
553         String strVal = this.strVal;
554         if (strVal != null) {
555             return strVal;
556         }
557 
558         final StringBuilder buf = new StringBuilder(96);
559 
560         // Prepend the current channel information if available.
561         final Channel ch = channel();
562         final boolean hasChannel = ch != null;
563         if (hasChannel) {
564             buf.append(ch);
565 
566             final InetAddress remote = ((InetSocketAddress) remoteAddress()).getAddress();
567             final InetAddress client = clientAddress();
568             if (remote != null && !remote.equals(client)) {
569                 buf.append("[C:").append(client.getHostAddress()).append(']');
570             }
571         }
572 
573         buf.append('[')
574            .append(sessionProtocol().uriText())
575            .append("://")
576            .append(virtualHost().defaultHostname());
577 
578         final InetSocketAddress laddr = localAddress();
579         if (laddr != null) {
580             buf.append(':').append(laddr.getPort());
581         } else {
582             buf.append(":-1"); // Port unknown.
583         }
584 
585         buf.append(path())
586            .append('#')
587            .append(method())
588            .append(']');
589 
590         strVal = buf.toString();
591         if (hasChannel) {
592             this.strVal = strVal;
593         }
594         return strVal;
595     }
